$\vec x_\xi(n) = \begin{pmatrix} x_n^{\xi_1} \\ \vdots \\ x_n^{\xi_m}  \end{pmatrix}, \vec y_\eta(n) = \begin{pmatrix} x_n^{\xi_1} \\ \vdots \\ x_n^{\xi_l}  \end{pmatrix} = Pr_l \circ \vec x_\xi$ 
Проектор $Pr_l: \begin{pmatrix} x_1 \\ \vdots \\ x_m  \end{pmatrix} \rightarrow \begin{pmatrix} x_1 \\ \vdots \\ x_l  \end{pmatrix} \equiv \begin{pmatrix} 1 &  & 0 & \cdots & 0 \\  & \ddots  \\ 0 &  & 1 & \cdots & 0 \end{pmatrix}_{l\times m}$ 
$\mathbb P_\xi := \mathbb P_N \circ \vec x_\xi^{-1} : \mathcal B(\mathbb R^m) \rightarrow [0, 1]$
$\mathbb P_\eta := \mathbb P_N \circ \vec y_\eta^{-1} : \mathcal B(\mathbb R^l) \rightarrow [0, 1]$ 

$(f \circ g)^{-1} (B) = (g^{-1} \circ f^{-1}) (B) \Rightarrow \vec y^{-1}_\eta = \vec x_\xi^{-1} \circ Pr_l^{-1} \Rightarrow \mathbb P_\eta = \mathbb P_\xi \circ Pr^{-1}_l$

$M = \{1, \cdots, m\}, L \subset M$
$x_\xi : n\rightarrow (x_n^{\xi_k})_{k \in M} \in \mathbb R^M$ 
$y_\eta : n\rightarrow (x_n^{\xi_k})_{k \in L} = x_\xi \upharpoonright _L \in \mathbb R^L$ 
$Pr_{M\rightarrow L}: \mathbb R^M \rightarrow \mathbb R^L$
$\mathbb P\eta = \mathbb P_\xi \circ Pr^{-1}_{M\rightarrow L}$
